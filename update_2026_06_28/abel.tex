\section{MDP Formulation}
% ============================================================

% ---- Slide 0: Define the MDP -------------------------------
\begin{frame}{Defining the Problem as an MDP}
  $$ \mathcal{M} = (\underbrace{\mathcal{S}}_{\text{state}},\ \underbrace{\mathcal{A}}_{\text{action}},\ \underbrace{P}_{\text{transition}},\ \underbrace{R}_{\text{reward}},\ \underbrace{\gamma}_{\text{discount}}) $$

  \vspace{0.6em}
  \begin{center}
  \renewcommand{\arraystretch}{1.25}
  \begin{tabular}{l c c}
    \toprule
    \textbf{Component} & \textbf{Earlier (v1)} & \textbf{Now (v2)} \\
    \midrule
    State      & $12$-dim aggregated   & \textcolor{SoftCoral}{\textbf{changed}} \\
    Action     & Discrete$(K{+}1)$ promote         & unchanged \\
    Transition & $\Delta t$ step, stochastic arrivals & unchanged \\
    Reward     & $\omega_u u_t - \omega_Q Q_t$ (unbounded)   & \textcolor{SoftCoral}{\textbf{changed}} \\
    Horizon    & truncate at $T$ or drained        & unchanged \\
    \bottomrule
  \end{tabular}
  \end{center}

  \vspace{0.3em}
  \centering
  \notebar{\mbox{$\gamma$ = discount} \;\(\mid\)\; \mbox{$\Delta t$ = sim step} \;\(\mid\)\; \mbox{$K$ = visible queued jobs} \;\(\mid\)\; \mbox{$T$ = horizon} \;\(\mid\)\; \mbox{$u_t$ = utilisation} \;\(\mid\)\; \mbox{$Q_t$ = queue length} \;\(\mid\)\; \mbox{$\omega_u,\omega_Q$ = reward weights}}
\end{frame}

% ---- Slide 2: State change (CHANGED) -----------------------
\begin{frame}{State $\;s_t$ \hfill \small\textcolor{SoftCoral}{(redesigned)}}
  \footnotesize
  \begin{columns}[T,onlytextwidth]
    \column{0.315\textwidth}
      \begin{block}{v1: aggregated summary}
        \begin{minipage}[t][4.0cm]{\linewidth}
        \begin{minipage}[t][2.0cm]{\linewidth}
        $s_t \in \mathbb{R}^{12}$: cluster stats $+$ queue means.
        $$
          s_t = \big[\,u_t,\ n_{\text{free}},\ Q_t,\ \bar{q}_t,\ \bar{w}_t,\ \dots\,\big]
        $$
        \end{minipage}
        \begin{itemize}\setlength\itemsep{3pt}
          \item Queue summarised by \emph{means} only
          \item No per-job detail
        \end{itemize}
        \end{minipage}
      \end{block}

    \hfill
    \column{0.315\textwidth}
      \begin{block}{v2(a): per-job features}
        \begin{minipage}[t][4.0cm]{\linewidth}
        \begin{minipage}[t][2.0cm]{\linewidth}
        $s_t \in \mathbb{R}^{2K+2}$, one entry per visible job:
        $$
          s_t = \Big[\tfrac{t}{T},\ \tfrac{n_{\text{free}}}{N},\
          \big(\tfrac{q_i}{N}\big)_{i=1}^{K},\
          \big(\tfrac{w_i}{T}\big)_{i=1}^{K}\Big]
        $$
        \end{minipage}
        \begin{itemize}\setlength\itemsep{3pt}
          \item Exposes \emph{each} visible job, not just a mean
          \item Empty slots padded with $-1$ sentinel
        \end{itemize}
        \end{minipage}
      \end{block}

    \hfill
    \column{0.315\textwidth}
      \begin{block}{v2(b): + priority}
        \begin{minipage}[t][4.0cm]{\linewidth}
        \begin{minipage}[t][2.0cm]{\linewidth}
        $s_t \in \mathbb{R}^{3K+2}$: adds one priority per visible job:
        $$
          s_t = \Big[\dots,\ \big(\tfrac{p_i}{10}\big)_{i=1}^{K}\Big]
        $$
        \end{minipage}
        \begin{itemize}\setlength\itemsep{3pt}
          \item Priority $p_i \in \{0,\dots,10\}$, normalised $p_i/10$
          \item Lets the policy weigh \emph{urgency}, not just size
        \end{itemize}
        \end{minipage}
      \end{block}
  \end{columns}

  \vspace{0.2em}
  \centering
  \small\textcolor{DeepBlue}{\textbf{Why:}} per-job observation lets the policy reason about \emph{which} job to promote, not just queue averages.
  \vspace{1em}
  
  \notebar{\mbox{$t,T$ = time, horizon} \;\(\mid\)\; \mbox{$N$ = total nodes}\;\(\mid\)\; \mbox{$n_{\text{free}}$ = idle nodes} \;\(\mid\)\; \mbox{$K$ = visible jobs} \;\(\mid\)\; \mbox{$u_t$ = utilisation} \;\(\mid\)\; \mbox{$Q_t$ = queue length} \;\(\mid\)\; \mbox{$\bar{q}_t,\bar{w}_t$ = mean size, wall time} \;\(\mid\)\; \mbox{$q_i,w_i$ = job $i$ size, wall time} \;\(\mid\)\; \mbox{$p_i$ = job $i$ priority}}
\end{frame}

% ---- Slide 5: Reward change (CHANGED) ----------------------
\begin{frame}{Reward $\;R(s_t, a_t)$ \hfill \small\textcolor{SoftCoral}{(redesigned)}}
  \footnotesize
  \begin{columns}[T,onlytextwidth]
    \column{0.315\textwidth}
        \begin{block}{v1: util.\ $-$ queue}
        \begin{minipage}[t][4.0cm]{\linewidth}
        $$ r_t = \omega_u u_t - \omega_Q Q_t $$
        \begin{itemize}\setlength\itemsep{2pt}
          \item Rewards busy nodes, penalises backlog
          \item Penalty uses \textbf{raw} queue count $Q_t$
          \item \alert{Unbounded}: scales with workload size
        \end{itemize}
        \end{minipage}
      \end{block}

    \hfill
    \column{0.315\textwidth}
      \begin{block}{v2(a): schedulable-idle}
        \begin{minipage}[t][4.0cm]{\linewidth}
        $$ r^{a}_t = -\omega_a\,\frac{\min(D_t, n_{\text{free}})}{N} $$
        \centering
        \begin{tikzpicture}[x=0.34cm,y=0.24cm]
          \def\demand{8.0}
          \def\offer{5.1}
          \def\sched{5.1}
          \draw[fill=SkyBlue!85,draw=DeepBlue!95!black,line width=0.3pt] (0,2.2) rectangle (\demand,2.9);
          \draw[fill=SoftCoral!85,draw=DeepBlue!95!black,line width=0.3pt] (0,1.3) rectangle (\offer,2.0);
          \draw[fill=WarmGold!85,draw=DeepBlue!95!black,line width=0.3pt] (0,0.4) rectangle (\sched,1.1);
          \draw[dashed,line width=0.35pt] (\sched,0.2) -- (\sched,3.1);
          \node[anchor=east,font=\tiny] at (-0.25,2.55) {Queue demand($D_t$)};
          \node[anchor=east,font=\tiny] at (-0.25,1.65) {Free capacity($n_{\text{free}}$)};
          \node[anchor=east,font=\tiny] at (-0.25,0.75) {Penalty};
        \end{tikzpicture}
        \vspace{-0.5em}
        \begin{itemize}\setlength\itemsep{1pt}\setlength\topsep{1pt}\setlength\parsep{0pt}
          \item Charges only for idle nodes the queue \emph{could} use
          \item Bounded $[-1,0]$, \textcolor{DeepBlue}{load-agnostic}
        \end{itemize}
        \end{minipage}
      \end{block}

    \hfill
    \column{0.315\textwidth}
      \begin{block}{v2(b): priority slowdown}
        \begin{minipage}[t][4.0cm]{\linewidth}
        $$ r^{b}_t =r^{a}_t -\omega_b\,\textstyle\sum_{j}\left( p_j +\tfrac{1}{w_j}\right) $$
        \begin{itemize}\setlength\itemsep{2pt}
          \item Over \textbf{queued} jobs
          \item Per-job term: $p_j + \tfrac{1}{w_j}$
          \item For equal priority, smaller wall-time gets larger score
        \end{itemize}
        \end{minipage}
      \end{block}
  \end{columns}

  \vspace{0.2em}
  \centering
  \textcolor{DeepBlue}{\textbf{Why:}} queue length over-penalises a full cluster and utilisation over-penalises an empty queue: a noisy, workload-dependent signal. Schedulable-idle charges only for avoidable idleness.
  \vspace{0.4em}
 
 \notebar{\mbox{$u_t$ = utilisation} \;\(\mid\)\; \mbox{$Q_t$ = queue length} \;\(\mid\)\; \mbox{$D_t$ =
  queue demand (nodes)} \;\(\mid\)\; \mbox{$n_{\text{free}}$ = idle nodes} \;\(\mid\)\; \mbox{$N$ = total nodes}
   \;\(\mid\)\; \mbox{$\Delta t$ = sim step} \;\(\mid\)\; \mbox{$p_j,w_j$ = job $j$ priority, wall time} \;\(\mid\)\; 
   \mbox{$\omega_u,\omega_Q,\omega_a,\omega_b$ = reward weights}}


\end{frame}


% ---- Slide 7: Why it matters -------------------------------
\begin{frame}{Why These Changes Matter}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \textbf{From v1 to v2 we gain:}
      \begin{itemize}
        \item \textbf{Bounded signals}: state in $[0,1]$, reward in $[-1,0]$
        \item \textbf{Load invariance}: far less sensitive to raw queue size and load level
        \item \textbf{Richer observation}: per-job features instead of a single mean
        \item \textbf{Direct objective}: penalise \emph{schedulable} idle capacity (head-of-line blocking)
      \end{itemize}

    \column{0.48\textwidth}
      \begin{alertblock}{Main result: training is now possible}
        Under the old design returns stayed \textbf{flat and noisy}: the agents
        \textbf{failed to learn}. With the redesigned state and reward, returns
        \textbf{climb steadily}: the agent \textbf{clearly learns} to promote
        jobs that fit the free nodes.
      \end{alertblock}
  \end{columns}
\end{frame}
